import { useRef, useState, useEffect } from "react";
import { BrowserMultiFormatReader } from "@zxing/browser";
 
const API_BASE = "http://localhost:3000"; // 你的 inventory-server 地址
 
export default function EquipmentPage({ onSaved }: { onSaved: (saved: any) => void }) {
  const videoRef = useRef<HTMLVideoElement | null>(null);
  const readerRef = useRef<BrowserMultiFormatReader | null>(null);
 
  const [barcodeText, setBarcodeText] = useState("");
  const [name, setName] = useState("");
  const [size, setSize] = useState("");
  const [price, setPrice] = useState("");
  const [matchedItemId, setMatchedItemId] = useState<number | null>(null);
  const [error, setError] = useState<string>("");
 
  // 自动启动摄像头
  useEffect(() => {
    const reader = new BrowserMultiFormatReader();
    readerRef.current = reader;
    let stopped = false;
 
    (async () => {
      try {
        const devices = await BrowserMultiFormatReader.listVideoInputDevices();
        const deviceId = devices?.[0]?.deviceId;
 
        await reader.decodeFromVideoDevice(
          deviceId,
          videoRef.current!,
          async (result) => {
            if (stopped || !result) return;
 
            const text = result.getText();
            setBarcodeText(text);
 
            // 1️⃣ 尝试调用数据库接口匹配条码
            try {
              const res = await fetch(`${API_BASE}/api/match`, {
                method: "POST",
                headers: { "Content-Type": "application/json" },
                body: JSON.stringify({ barcode: text }),
              });
 
              if (res.ok) {
                const data = await res.json();
                if (data?.match) {
                  setName(data.match.name || "");
                  setSize(data.match.size || "");
                  setPrice(data.match.price_cents != null ? (data.match.price_cents / 100).toString() : "");
                  setMatchedItemId(data.match.id);
                  return; // 成功匹配，停止 JSON 自动解析
                }
              }
            } catch (err) {
              console.warn("API match error:", err);
            }
 
            // 2️⃣ 如果不是数据库条码，则尝试解析 JSON
            try {
              const obj = JSON.parse(text);
              setName(obj.name || "");
              setSize(obj.size || "");
              setPrice(obj.price != null ? String(obj.price) : "");
            } catch {}
          }
        );
      } catch (err) {
        setError(err instanceof Error ? err.message : String(err));
      }
    })();
 
    return () => {
      stopped = true;
      readerRef.current?.reset();
    };
  }, []);
 
  // 保存到数据库或 FHIR
  const onSave = async () => {
    if (!name) { setError("请先扫描或填写商品名称"); return; }
 
    try {
      if (matchedItemId) {
        await fetch(`${API_BASE}/api/usage`, {
          method: "POST",
          headers: { "Content-Type": "application/json" },
          body: JSON.stringify({ item_id: matchedItemId, qty: 1 }),
        });
      }
      alert("保存成功！");
      onSaved && onSaved({ name, size, price });
      setName(""); setSize(""); setPrice(""); setBarcodeText(""); setMatchedItemId(null);
    } catch (err) {
      setError(err instanceof Error ? err.message : String(err));
    }
  };
 
  return (
    <div>
      <video ref={videoRef} width={400} height={300} autoPlay muted />
      <div>
        <label>扫描结果: {barcodeText}</label>
        <label>名称: <input value={name} onChange={e => setName(e.target.value)} /></label>
        <label>大小: <input value={size} onChange={e => setSize(e.target.value)} /></label>
        <label>价格: <input value={price} onChange={e => setPrice(e.target.value)} /></label>
        <button onClick={onSave}>保存</button>
        {error && <div style={{color:"red"}}>{error}</div>}
      </div>
    </div>
  );
}
